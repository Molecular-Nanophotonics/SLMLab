Given the electromagnetic field $E(x, y, 0)$ defined in the plane $z = 0$, the field $E(x,y,z)$ in the plane $z$ can be calculated using the Fresnel diffraction integral:
\begin{align}
E(x,y,z) = \frac{\mathrm e^{\i kz}}{\i\lambda z}\iint\limits_{-\infty}^{+\infty} E(x', y', 0)\,\e^{\i\frac{k}{2z}\left((x-x')^2 + (y-y')^2\right)\;}\mathrm dx'\mathrm dy'
\end{align}

\begin{align}
E(x,y,z) = E(x, y, 0) 
* h(x,y,z)\ ,
\end{align}
where
\begin{align}
h(x, y, z) = \frac{\mathrm e^{\i kz}}{\i\lambda z}\e^{\i\frac{k}{2z}(x^2 + y^2)}
\end{align}
\begin{align}
(f * g)(x) = \int f(x)\,g(x - x')\,\mathrm dx' 
\end{align}

Since ${\mathcal {F}}\{f*g\}={\mathcal {F}}\{f\}\cdot {\mathcal {F}}\{g\}$
\begin{align}
\check{E}(f_x, f_y, z) = \check{E}(f_x, f_y, 0)H(f_x, f_y, z)
\end{align}

\begin{align}
H(f_x, f_y, z) &= \mathcal F\{h(x, y,z)\} \\
&= \frac{\mathrm e^{\i kz}}{\i\lambda z} \mathcal F\left\{\e^{\i\frac{k}{2z}x^2 }\right\}\mathcal F\left\{\e^{\i\frac{k}{2z}y^2 }\right\}
\end{align}

\begin{align}
\mathcal F\left\{\e^{\i\alpha t^2 }\right\} = \sqrt{\frac{\pi}{\alpha}} \,\e^{\i\frac{\pi}{4}}\,\e^{-\i\frac{(2\pi f)^2}{4\alpha}}
\end{align}

\begin{align}
H(f_x, f_y, z) &= \frac{\mathrm e^{\i kz}}{\i\lambda z}\sqrt{\frac{2\pi z}{k}} \,\e^{\i\frac{\pi}{4}}\,\e^{-\i\frac{4\pi^2 f_x^2 2z}{4k}} \sqrt{\frac{2\pi z}{k}} \,\e^{\i\frac{\pi}{4}}\,\e^{-\i\frac{4\pi^2 f_y^2 2z}{4k}} \\
&=  \frac{\mathrm e^{\i kz}}{\i\lambda z}\frac{2\pi z}{k}\,\e^{-\i\pi\lambda z f_x^2}\,\e^{-\i\pi\lambda z f_y^2} = \e^{\i kz}\e^{-\i\pi\lambda z(f_x^2 + f_y^2)}
\end{align}

\begin{align}
H(f_x, f_y, z) = \e^{\i kz}\e^{-\i\pi\lambda z(f_x^2+f_y^2)}
\end{align}

\begin{align}
\check{E}(f_x, f_y, z) = \e^{\i kz}\e^{-\i\pi\lambda z(f_x^2+f_y^2)}\,\check{E}(f_x, f_y, 0)
\end{align}

Propagate from the SLM plane $E_\mathrm{SLM}(x, y)$ to the lense:
\begin{align}
E_\mathrm{Lense}^-(x,y) = \frac{\mathrm e^{\i kf}}{\i\lambda f}\e^{\i\frac{k}{2f}(x^2+y^2)}\iint\limits_{-\infty}^{+\infty} E_\mathrm{SLM}(x', y')\,\e^{\i\frac{k}{2f}(x'^2 + y'^2)}\e^{-\i\frac{2\pi}{\lambda f}(xx'+yy')}\;\mathrm dx'\mathrm dy'
\end{align}

\begin{align}
E_\mathrm{Lens}^+(x,y) = E_\mathrm{Lens}^-(x,y)\e^{-\i\frac{k}{2f}(x^2+y^2)}
\end{align}

Propagate from $E_\mathrm{Lense}^+(x,y)$ to the focal plane:
\begin{align}
E_\mathrm{Focus}(x,y) = \frac{\mathrm e^{\i kf}}{\i\lambda f}\e^{\i\frac{k}{2f}(x^2+y^2)}\iint\limits_{-\infty}^{+\infty} E_\mathrm{Lens}^+(x', y')\,\e^{\i\frac{k}{2f}(x'^2 + y'^2)}\e^{-\i\frac{2\pi}{\lambda f}(xx'+yy')}\;\mathrm dx'\mathrm dy'
\end{align}

...:
\begin{align}
E_\mathrm{Focus}(x,y) &= \frac{\mathrm e^{\i kf}}{\i\lambda f}\e^{\i\frac{k}{2f}(x^2+y^2)}\iint\limits_{-\infty}^{+\infty} E_\mathrm{Lens}^-(x', y')\,\e^{-\i\frac{2\pi}{\lambda f}(xx'+yy')}\;\mathrm dx'\mathrm dy' \\ 
&= \frac{\mathrm e^{\i kf}}{\i\lambda f}\e^{\i\frac{k}{2f}(x^2+y^2)}\,\,4\pi^2\,\check{E}_\mathrm{Lens}^{-}(f_x, f_y)
\end{align}

\begin{align}
\check{E}_\mathrm{Lens}^{-}(f_x, f_y) = \e^{\i kf}\e^{-\i\pi\lambda f(f_x^2+f_y^2)}\,4\pi^2\,\check{E}_\mathrm{SLM}^{-}(f_x, f_y)
\end{align}


$f_x = x/(\lambda f)$, $f_y = y/(\lambda f)$
\begin{align}
E_\mathrm{Focus}(x,y) =  \frac{\mathrm e^{2\i kf}}{\i\lambda f}\,4\pi^2\,\check{E}_\mathrm{SLM}(f_x, f_y)
\end{align}
